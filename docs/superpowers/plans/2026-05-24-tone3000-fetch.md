```rust
//! Supabase PostgREST client for tone3000.com.
//!
//! The anon JWT below is the same JWT the public website embeds in
//! its JS bundle (role: anon, project ref gzybiuopxkdxbytnojds, exp
//! 2035-02-06). It is *not* a user secret — it is the project's
//! intended public API key. If Supabase rotates it, `refresh_anon_key`
//! re-extracts the current value from the live homepage.

use anyhow::{anyhow, Context, Result};
use serde::{Deserialize, Serialize};
use std::io::Write;
use std::path::PathBuf;
use std::time::Duration;

pub const ANON_KEY_DEFAULT: &str = "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6Imd6eWJpdW9weGtkeGJ5dG5vamRzIiwicm9sZSI6ImFub24iLCJpYXQiOjE3MzgwODIxNjUsImV4cCI6MjA1MzY1ODE2NX0.Gq66BJXjtLsqP2nAGXm9Xb9PAjoeZalWUj66K4nmVSU";
pub const REST_BASE: &str = "https://api.tone3000.com/rest/v1";
pub const SITE_BASE: &str = "https://www.tone3000.com";

#[derive(Debug, Clone, Serialize)]
pub struct SearchFilters {
    pub query_term: String,
    pub page_number: u32,
    pub page_size: u32,
    pub order_by: String,                       // "newest" | "trending" | ...
    pub tag_names: Option<Vec<String>>,
    pub make_names: Option<Vec<String>>,
    pub gear_filters: Option<Vec<String>>,      // "ir" | "amp" | "pedal" | "full-rig" | "outboard"
    pub is_calibrated: bool,
    pub size_filters: Option<Vec<String>>,
    pub usernames: Option<Vec<String>>,
}

impl SearchFilters {
    pub fn latest(page_size: u32) -> Self {
        Self {
            query_term: String::new(),
            page_number: 1,
            page_size,
            order_by: "newest".into(),
            tag_names: None,
            make_names: None,
            gear_filters: None,
            is_calibrated: false,
            size_filters: None,
            usernames: None,
        }
    }
    pub fn query(q: &str, page_size: u32) -> Self {
        Self { query_term: q.into(), ..Self::latest(page_size) }
    }
}

#[derive(Debug, Clone, Deserialize)]
pub struct SearchRow {
    pub id: u64,
    pub title: String,
    #[serde(default)]
    pub description: String,
    #[serde(default)]
    pub tags: Vec<String>,
    #[serde(default)]
    pub makes: Vec<String>,
    pub gear: String,
    pub platform: String,                        // "nam" | "ir"
    pub models_count: u32,
    pub created_at: String,
    pub published_at: Option<String>,
    pub username: Option<String>,
    pub total_count: Option<u32>,
}

#[derive(Debug, Clone, Deserialize)]
pub struct Tone {
    pub id: u64,
    pub title: String,
    #[serde(default)]
    pub description: String,
    pub gear: String,
    #[serde(default)]
    pub is_public: bool,
    #[serde(default)]
    pub links: serde_json::Value,
    #[serde(default)]
    pub images: Option<Vec<String>>,
    pub user_id: String,
    pub created_at: String,
}

#[derive(Debug, Clone, Deserialize)]
pub struct Model {
    pub id: u64,
    pub name: String,
    pub model_url: String,
    pub size: Option<String>,
    pub position: Option<String>,
    pub architecture_version: Option<String>,
    pub created_at: String,
}

pub fn parse_search_response(raw: &str) -> Result<Vec<SearchRow>> {
    serde_json::from_str(raw).context("parse search_tones_a2 response")
}
pub fn parse_models_response(raw: &str) -> Result<Vec<Model>> {
    serde_json::from_str(raw).context("parse /rest/v1/models response")
}
pub fn parse_tone_response(raw: &str) -> Result<Tone> {
    let v: Vec<Tone> = serde_json::from_str(raw).context("parse /rest/v1/tones response")?;
    v.into_iter().next().ok_or_else(|| anyhow!("tone not found"))
}

pub struct Client {
    http: reqwest::blocking::Client,
    anon_key: String,
}

impl Client {
    pub fn new() -> Result<Self> {
        let anon_key = resolve_anon_key()?;
        let http = reqwest::blocking::Client::builder()
            .user_agent("OpenRig-plugins/tone3000_fetch")
            .timeout(Duration::from_secs(60))
            .build()?;
        Ok(Self { http, anon_key })
    }

    fn auth_headers(&self) -> reqwest::header::HeaderMap {
        use reqwest::header::{HeaderMap, HeaderValue, AUTHORIZATION};
        let mut h = HeaderMap::new();
        h.insert("apikey", HeaderValue::from_str(&self.anon_key).unwrap());
        h.insert(AUTHORIZATION, HeaderValue::from_str(&format!("Bearer {}", self.anon_key)).unwrap());
        h.insert("content-profile", HeaderValue::from_static("public"));
        h
    }

    pub fn search(&self, filters: &SearchFilters) -> Result<Vec<SearchRow>> {
        let url = format!("{}/rpc/search_tones_a2", REST_BASE);
        let body = serde_json::to_string(filters)?;
        let resp = self.http.post(&url).headers(self.auth_headers())
            .header(reqwest::header::CONTENT_TYPE, "application/json")
            .body(body).send()?;
        let status = resp.status();
        let text = resp.text()?;
        if !status.is_success() {
            return Err(anyhow!("search_tones_a2 HTTP {}: {}", status, text));
        }
        parse_search_response(&text)
    }

    pub fn tone(&self, id: u64) -> Result<Tone> {
        let url = format!("{}/tones?id=eq.{}&select=*", REST_BASE, id);
        let text = self.http.get(&url).headers(self.auth_headers()).send()?.error_for_status()?.text()?;
        parse_tone_response(&text)
    }

    pub fn models(&self, tone_id: u64) -> Result<Vec<Model>> {
        let url = format!(
            "{}/models?tone_id=eq.{}&select=id,name,model_url,size,position,architecture_version,created_at",
            REST_BASE, tone_id
        );
        let text = self.http.get(&url).headers(self.auth_headers()).send()?.error_for_status()?.text()?;
        parse_models_response(&text)
    }

    pub fn download(&self, model_url: &str, dest: &std::path::Path) -> Result<u64> {
        let resp = self.http.get(model_url).send()?.error_for_status()?;
        let bytes = resp.bytes()?;
        let len = bytes.len() as u64;
        std::fs::create_dir_all(dest.parent().unwrap_or(std::path::Path::new(".")))?;
        let mut f = std::fs::File::create(dest)?;
        f.write_all(&bytes)?;
        Ok(len)
    }
}

fn resolve_anon_key() -> Result<String> {
    if let Ok(k) = std::env::var("TONE3000_ANON_KEY") {
        return Ok(k);
    }
    if let Some(p) = cached_token_path() {
        if let Ok(k) = std::fs::read_to_string(&p) {
            return Ok(k.trim().to_string());
        }
    }
    Ok(ANON_KEY_DEFAULT.into())
}

fn cached_token_path() -> Option<PathBuf> {
    dirs::config_dir().map(|d| d.join("openrig").join("tone3000.token"))
}

/// Re-extract the anon JWT from the live tone3000 homepage. Persists
/// the result to `~/.config/openrig/tone3000.token` if the embedded
/// `ANON_KEY_DEFAULT` ever stops working.
pub fn refresh_anon_key() -> Result<String> {
    let http = reqwest::blocking::Client::builder()
        .user_agent("OpenRig-plugins/tone3000_fetch")
        .timeout(Duration::from_secs(30))
        .build()?;
    let html = http.get(SITE_BASE).send()?.error_for_status()?.text()?;
    let re = regex::Regex::new(r#"eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9\.[A-Za-z0-9_-]+\.[A-Za-z0-9_-]+"#)?;
    let mut hits: Vec<&str> = re.find_iter(&html).map(|m| m.as_str()).collect();
    // The page also embeds the user's access_token if logged in; we
    // want the longest anon JWT, which is the static one rendered into
    // every Supabase client config.
    hits.sort_by_key(|s| std::cmp::Reverse(s.len()));
    let key = hits.into_iter().next().ok_or_else(|| anyhow!("no JWT found in homepage; tone3000 layout changed?"))?.to_string();
    if let Some(p) = cached_token_path() {
        std::fs::create_dir_all(p.parent().unwrap())?;
        std::fs::write(&p, &key)?;
    }
    Ok(key)
}
```

```rust
//! Walks plugins/source/{ir,nam}/**/manifest.yaml and indexes every
//! tone3000 URL that already appears under `sources:`. Used to mark
//! search results as already-imported vs new.

use anyhow::{Context, Result};
use serde::Deserialize;
use std::collections::HashMap;
use std::path::{Path, PathBuf};

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum LocalStatus {
    New,
    Imported,
}

#[derive(Debug, Default)]
pub struct LocalIndex {
    /// tone id -> manifest path
    pub by_id: HashMap<u64, PathBuf>,
}

impl LocalIndex {
    pub fn status(&self, tone_id: u64) -> LocalStatus {
        if self.by_id.contains_key(&tone_id) { LocalStatus::Imported } else { LocalStatus::New }
    }
}

#[derive(Deserialize)]
struct PartialManifest {
    #[serde(default)]
    sources: Vec<String>,
}

pub fn scan_repo(repo_root: &Path) -> Result<LocalIndex> {
    let mut idx = LocalIndex::default();
    let source_root = repo_root.join("plugins").join("source");
    if !source_root.exists() { return Ok(idx); }
    for entry in walkdir::WalkDir::new(&source_root).into_iter().filter_map(|e| e.ok()) {
        if !entry.file_type().is_file() { continue; }
        if entry.file_name() != "manifest.yaml" { continue; }
        let raw = std::fs::read_to_string(entry.path()).with_context(|| format!("read {}", entry.path().display()))?;
        let pm: PartialManifest = match serde_yaml::from_str(&raw) {
            Ok(v) => v,
            Err(_) => continue,
        };
        for src in pm.sources {
            if let Some(id) = parse_tone3000_id(&src) {
                idx.by_id.insert(id, entry.path().to_path_buf());
            }
        }
    }
    Ok(idx)
}

fn parse_tone3000_id(url: &str) -> Option<u64> {
    // Accept https://www.tone3000.com/tones/<id>[anything]
    let prefix = "tone3000.com/tones/";
```

```rust
//! Best-effort parameter axis classifier for tone3000 capture names.
//!
//! Returns `(axis_name, normalized_value)` tuples. Unknown tokens are
//! dropped — the scaffolder emits a `# TODO` YAML comment so the user
//! sees the miss.

const MICS: &[(&str, &str)] = &[
    ("sm57", "sm57"), ("sm7b", "sm7b"), ("md421", "md421"), ("re20", "re20"),
    ("beta52", "beta52"), ("c414", "c414"), ("r121", "r121"), ("r10", "r10"),
    ("m160", "m160"), ("u87", "u87"), ("royer", "r121"), ("royer-r10", "r10"),
];

const POSITIONS: &[(&str, &str)] = &[
    ("cap edge", "cap_edge"), ("cone edge", "cone_edge"),
    ("cap", "cap"), ("cone", "cone"), ("distant", "distant"),
    ("12 inch", "12_inch"), ("12in", "12_inch"), ("24in", "24_inch"),
];

const SPEAKERS: &[(&str, &str)] = &[
    ("upper", "upper"), ("lower", "lower"),
];

const VOICINGS: &[(&str, &str)] = &[
    ("hgt", "hgt"), ("hg", "hg"), ("normal", "normal"),
    ("clean", "clean"), ("crunch", "crunch"), ("lead", "lead"),
];

pub fn classify_capture(name: &str) -> Vec<(&'static str, String)> {
    let lower = name.to_ascii_lowercase();
    let mut out: Vec<(&'static str, String)> = Vec::new();

    for (needle, value) in POSITIONS { if lower.contains(needle) { push_unique(&mut out, "position", (*value).into()); break; } }
    for (needle, value) in MICS      { if lower.contains(needle) { push_unique(&mut out, "mic", (*value).into()); break; } }
    for (needle, value) in SPEAKERS  { if lower.contains(needle) { push_unique(&mut out, "speaker", (*value).into()); break; } }
    for (needle, value) in VOICINGS  { if lower.contains(needle) { push_unique(&mut out, "voicing", (*value).into()); break; } }

    // gain/mids/bass/treble as "<g|m|b|t><digit>" e.g. "5g", "g5", "7m", "mids 5"
    let re_axis = regex::Regex::new(r"\b(gain|mids?|bass|treble)\s*[:=]?\s*(\d+)\b").unwrap();
    for c in re_axis.captures_iter(&lower) {
        let axis = match &c[1] { "mids" | "mid" => "mids", other => match other { "gain" => "gain", "bass" => "bass", "treble" => "treble", _ => "?" } };
        if axis != "?" { push_unique(&mut out, axis_to_static(axis), c[2].to_string()); }
    }
    let re_short = regex::Regex::new(r"\b(\d+)\s*(g|m|b|t)\b").unwrap();
    for c in re_short.captures_iter(&lower) {
        let axis = match &c[2] { "g" => "gain", "m" => "mids", "b" => "bass", "t" => "treble", _ => "?" };
        if axis != "?" { push_unique(&mut out, axis_to_static(axis), c[1].to_string()); }
    }

    out
}

fn push_unique(out: &mut Vec<(&'static str, String)>, axis: &'static str, value: String) {
    if !out.iter().any(|(a, _)| *a == axis) { out.push((axis, value)); }
}

fn axis_to_static(s: &str) -> &'static str {
    match s {
        "mic" => "mic", "speaker" => "speaker", "position" => "position",
        "voicing" => "voicing", "gain" => "gain", "mids" => "mids",
        "bass" => "bass", "treble" => "treble",
        _ => "?",
    }
}

/// Aggregate per-axis distinct values across a list of capture names —
/// used by `scaffold.rs` to populate the `parameters:` block.
pub fn extract_axes(names: &[String]) -> Vec<(String, Vec<String>)> {
    use std::collections::{BTreeMap, BTreeSet};
    let mut acc: BTreeMap<&'static str, BTreeSet<String>> = BTreeMap::new();
    for n in names {
        for (axis, value) in classify_capture(n) {
            acc.entry(axis).or_default().insert(value);
        }
    }
    acc.into_iter().map(|(k, v)| (k.to_string(), v.into_iter().collect())).collect()
}
```

```rust
//! Generates draft `plugins/source/<kind>/<slug>/manifest.yaml` for an
//! imported tone3000 pack. Never overwrites — caller must ensure the
//! target dir does not exist.

use anyhow::{anyhow, Context, Result};
use std::collections::BTreeMap;
use std::path::{Path, PathBuf};

use crate::api::{Model, Tone};
use crate::infer::{classify_capture, extract_axes};

pub fn slug_for(brand: &str, model_name: &str) -> String {
    fn slugify(s: &str) -> String {
        let mut out = String::with_capacity(s.len());
        let mut last_us = true;
        for c in s.chars() {
            if c.is_ascii_alphanumeric() {
                out.push(c.to_ascii_lowercase());
                last_us = false;
            } else if !last_us {
                out.push('_');
                last_us = true;
            }
        }
        out.trim_matches('_').to_string()
    }
    let b = slugify(brand);
    let m = slugify(model_name);
    if m.starts_with(&b) || b.is_empty() { m } else { format!("{}_{}", b, m) }
}

pub fn unique_slug(repo_root: &Path, kind: &str, candidate: &str) -> String {
    let parent = repo_root.join("plugins").join("source").join(kind);
    let mut s = candidate.to_string();
    let mut n = 2u32;
    while parent.join(&s).exists() {
        s = format!("{}_{}", candidate, n);
        n += 1;
    }
    s
}

pub fn kind_for_platform(platform: &str) -> &'static str {
    match platform { "nam" => "nam", _ => "ir" }
}

pub fn build_manifest_yaml(
    tone: &Tone,
    models: &[Model],
    kind: &str,                 // "ir" | "nam"
    makes: &[String],
    slug: &str,
) -> String {
    let brand = makes.first().map(|m| first_word_lower(m)).unwrap_or_else(|| "unknown".into());
    let type_kw = match (kind, tone.gear.as_str()) {
        ("nam", "amp") => "amp",
        ("nam", "pedal") => "gain_pedal",
        ("ir", _) => "cab",
        _ => "amp",
    };
    let backend = if kind == "ir" { "ir" } else { "nam" };

    // captures + per-axis discovery
    let names: Vec<String> = models.iter().map(|m| m.name.clone()).collect();
    let axes = extract_axes(&names);

    let subdir = if kind == "ir" { "ir" } else { "captures" };

    let mut yaml = String::new();
    yaml.push_str("manifest_version: 1\n");
    yaml.push_str(&format!("id: {}_{}\n", kind, slug));
    yaml.push_str(&format!("display_name: {}\n", tone.title.trim()));
    yaml.push_str(&format!("brand: {}\n", brand));
    if makes.len() > 1 {
        yaml.push_str(&format!("# TODO: brand guessed from makes={:?}; pick the right one\n", makes));
    }
    yaml.push_str("sources:\n");
    yaml.push_str(&format!("- https://www.tone3000.com/tones/{}\n", tone.id));
    yaml.push_str(&format!("type: {}\n", type_kw));
    yaml.push_str(&format!("backend: {}\n", backend));
    if !axes.is_empty() {
        yaml.push_str("parameters:\n");
        for (axis, values) in &axes {
            yaml.push_str(&format!("- name: {}\n", axis));
            yaml.push_str(&format!("  display_name: {}\n", title_case(axis)));
            yaml.push_str("  values:\n");
            for v in values { yaml.push_str(&format!("  - {}\n", v)); }
        }
    }
    yaml.push_str("captures:\n");
    for m in models {
        let classified: BTreeMap<&'static str, String> = classify_capture(&m.name).into_iter().collect();
        let filename = file_basename(&m.model_url);
        yaml.push_str("- values:\n");
        if classified.is_empty() {
            yaml.push_str(&format!("  # TODO: could not infer parameter axes for {:?}\n", m.name));
        } else {
            for (a, v) in &classified { yaml.push_str(&format!("    {}: {}\n", a, v)); }
        }
        yaml.push_str(&format!("  file: {}/{}\n", subdir, filename));
    }
    yaml
}

fn first_word_lower(s: &str) -> String {
    s.split_whitespace().next().unwrap_or("").to_ascii_lowercase()
}

fn title_case(s: &str) -> String {
    let mut c = s.chars();
    match c.next() {
        Some(f) => f.to_ascii_uppercase().to_string() + c.as_str(),
        None => String::new(),
    }
}

fn file_basename(url: &str) -> String {
    url.rsplit('/').next().unwrap_or("file.bin").to_string()
}

/// End-to-end: download every model into `<target>/<subdir>/`, write
/// `<target>/manifest.yaml`. Refuses to write if `target` exists.
pub fn import_pack(
    client: &crate::api::Client,
    tone: &Tone,
    models: &[Model],
    makes: &[String],
    kind: &str,
    target_dir: &Path,
) -> Result<PathBuf> {
    if target_dir.exists() {
        return Err(anyhow!("refusing to overwrite existing {}", target_dir.display()));
    }
    let subdir = if kind == "ir" { "ir" } else { "captures" };
    std::fs::create_dir_all(target_dir.join(subdir))?;
    for m in models {
        let dest = target_dir.join(subdir).join(file_basename(&m.model_url));
        client.download(&m.model_url, &dest).with_context(|| format!("download {}", m.model_url))?;
    }
    let slug = target_dir.file_name().unwrap().to_string_lossy().into_owned();
    let yaml = build_manifest_yaml(tone, models, kind, makes, &slug);
    let manifest_path = target_dir.join("manifest.yaml");
    std::fs::write(&manifest_path, yaml)?;
    Ok(manifest_path)
}
```

```markdown
# tone3000 fetch — import workflow

Orchestrates `tools/tone3000_fetch` to bring an IR or NAM pack from
tone3000.com into `plugins/source/<kind>/<slug>/`. The binary is the
mechanism; this skill enforces the surrounding repo discipline.

## When this skill applies

User says any of:
- "what's new on tone3000?" / "latest packs" / "novidades do tone3000"
- "import tone3000 <id>" / "traz o pack <id>"
- "procura IR de <amp/cab>" / "search tone3000 for <X>"

## Required preconditions (LAW)

Before any download:

1. **There must be an issue.** If none exists, run
   `gh issue create --title "import tone3000 <id>: <pack name>" --body "<link to tone>"`
   and use the returned number as `<N>`.
2. **Isolated workspace.** `git worktree add .solvers/issue-<N> -b feature/issue-<N> main`
   then `cd .solvers/issue-<N>`. Never touch the user's checkout.
3. **Comment the plan on the issue** before downloading.

## Discovery commands

```
cargo run -q -p tone3000_fetch -- latest --limit 20
cargo run -q -p tone3000_fetch -- search "<query>" --limit 20
cargo run -q -p tone3000_fetch -- show <id>
```

Output marks each row `new` or `imported` based on `sources:` URLs
already present in `plugins/source/**/manifest.yaml`. Always prefer
`imported` packs over re-importing.

## Import command

```
cargo run -q -p tone3000_fetch -- import <id> [--kind ir|nam] [--slug <name>]
```

The binary:
- creates `plugins/source/<kind>/<slug>/` and `<subdir>/` (ir/ or captures/);
- downloads every model URL into `<subdir>/`;
- writes a draft `manifest.yaml` with `sources:` set and parameters
  inferred from capture names;
- emits `# TODO:` YAML comments where inference failed.

## After import (LAW — gate green before push)

1. **Resolve every `# TODO:` line** in the generated manifest. Inference
   misses are not optional — leaving them is a defect that ships.
2. **Run the gate** in the worktree:

   ```
   cargo build --release -p loudness-audit --bin qa_audit
   cargo run --release --bin pack_plugins
   ```

   Exit 0 / `0 failed` is required. A red gate here = a red OpenRig
   release (issue #5, #12).
3. **Comment the issue** with commit hash + files touched + gate result
   on every push.
4. **PR targets `main`.** Open only on explicit user request.

## Anti-patterns

```
❌ Running `import` in the user's main checkout (must be in .solvers/issue-N/)
❌ Pushing with `# TODO` comments still in the generated manifest
❌ Setting QA_AUDIT_SKIP=1 to silence a qa_audit failure on an imported pack
❌ Two tone packs in one PR — one issue = one tone
❌ Editing an existing manifest by re-running `import` (the binary refuses;
   that refusal is a feature, do not work around it)
